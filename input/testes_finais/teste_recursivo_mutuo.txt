    se n = 0 entao

inicio
    se n = 0 entao
        eh_impar := falso
    senao
        n := n - 1;
        eh_impar := eh_par
fim;

    n := 7;
    se eh_par entao
        escreva(1)
    senao
        escreva(0)
fim.
